```sql id="98x0pf"
CREATE EXTENSION IF NOT EXISTS vector;

CREATE TABLE document_chunks (
    id BIGSERIAL PRIMARY KEY,

    tenant_id UUID NOT NULL,

    document_id UUID NOT NULL,

    document_name TEXT NOT NULL,

    chunk_index INT NOT NULL,

    content TEXT NOT NULL,

    metadata JSONB DEFAULT '{}',

    embedding VECTOR(1536),

    created_at TIMESTAMP DEFAULT NOW(),

    updated_at TIMESTAMP DEFAULT NOW()
);

-- =====================================================
-- HNSW INDEX
-- =====================================================

CREATE INDEX idx_embedding_hnsw
ON document_chunks
USING hnsw (embedding vector_cosine_ops);

-- =====================================================
-- METADATA INDEXES
-- =====================================================

CREATE INDEX idx_tenant
ON document_chunks (tenant_id);

CREATE INDEX idx_document
ON document_chunks (document_id);

CREATE INDEX idx_metadata
ON document_chunks
USING gin(metadata);

-- =====================================================
-- FULL TEXT SEARCH
-- =====================================================

ALTER TABLE document_chunks
ADD COLUMN content_tsv tsvector
GENERATED ALWAYS AS (
    to_tsvector('english', content)
) STORED;

CREATE INDEX idx_fts
ON document_chunks
USING gin(content_tsv);
```

# Production Embedding Service

## app/embeddings/embedding_service.py

```python id="hbyndd"
import hashlib
import json

from openai import OpenAI

from app.core.config import settings
from app.db.redis_client import redis_client


client = OpenAI(
    api_key=settings.OPENAI_API_KEY,
)


class EmbeddingService:
    def __init__(self):
        self.model = settings.EMBEDDING_MODEL

    def _cache_key(self, text: str):
        return hashlib.md5(text.encode()).hexdigest()

    def create_embedding(self, text: str):
        """
        Production embedding generation
        with Redis caching.
        """

        cache_key = self._cache_key(text)

        cached = redis_client.get(cache_key)

        if cached:
            return json.loads(cached)

        response = client.embeddings.create(
            model=self.model,
            input=text,
        )

        embedding = response.data[0].embedding

        redis_client.setex(
            cache_key,
            86400,
            json.dumps(embedding),
        )

        return embedding

    def batch_create_embeddings(
        self,
        texts: list[str],
    ):
        """
        Batch embeddings for throughput optimization.
        """

        response = client.embeddings.create(
            model=self.model,
            input=texts,
        )

        return [
            item.embedding
            for item in response.data
        ]
```

---

# Why Batch Embeddings Matter

# 4. PRODUCTION VECTOR SEARCH

## app/retrieval/retriever.py

```python id="mzg8jc"
from sqlalchemy import text

from app.db.postgres import engine
from app.embeddings.embedding_service import (
    EmbeddingService,
)


class Retriever:
    def __init__(self):
        self.embedding_service = EmbeddingService()

    def retrieve(
        self,
        query: str,
        tenant_id: str,
        top_k: int = 10,
        similarity_threshold: float = 0.75,
        metadata_filters: dict | None = None,
    ):
        """
        Production-grade vector retrieval.
        """

        query_embedding = (
            self.embedding_service
            .create_embedding(query)
        )

        filter_clauses = [
            "tenant_id = :tenant_id"
        ]

        params = {
            "embedding": query_embedding,
            "tenant_id": tenant_id,
            "top_k": top_k,
            "threshold": similarity_threshold,
        }

        if metadata_filters:
            for key, value in metadata_filters.items():
                clause = (
                    f\"metadata->>'{key}' = :{key}\"
                )

                filter_clauses.append(clause)

                params[key] = value

        where_clause = " AND ".join(
            filter_clauses
        )

        sql = text(
            f"""
            SELECT
                id,
                document_name,
                chunk_index,
                content,
                metadata,
                1 - (
                    embedding <=> :embedding
                ) AS similarity_score

            FROM document_chunks

            WHERE
                {where_clause}

            AND
                1 - (
                    embedding <=> :embedding
                ) > :threshold

            ORDER BY embedding <=> :embedding

            LIMIT :top_k
            """
        )

        with engine.begin() as conn:
            rows = conn.execute(
                sql,
                params,
            ).fetchall()

        return [
            {
                "id": row[0],
                "document_name": row[1],
                "chunk_index": row[2],
                "content": row[3],
                "metadata": row[4],
                "similarity_score": row[5],
            }
            for row in rows
        ]
```

# Production Hybrid Search

## app/retrieval/hybrid_search.py

```python id="lsj6ur"
from sqlalchemy import text

from app.db.postgres import engine


class HybridSearch:
    def hybrid_retrieve(
        self,
        query_embedding,
        query_text,
        top_k=10,
    ):
        """
        Combines:
        - vector similarity
        - BM25 full-text search
        """

        sql = text(
            """
            WITH semantic_search AS (
                SELECT
                    id,
                    content,
                    1 - (
                        embedding <=> :embedding
                    ) AS semantic_score
                FROM document_chunks
                ORDER BY embedding <=> :embedding
                LIMIT 20
            ),

            keyword_search AS (
                SELECT
                    id,
                    content,
                    ts_rank(
                        content_tsv,
                        plainto_tsquery(:query)
                    ) AS keyword_score
                FROM document_chunks
                WHERE content_tsv @@ plainto_tsquery(:query)
                LIMIT 20
            )

            SELECT
                COALESCE(
                    s.id,
                    k.id
                ) AS id,

                COALESCE(
                    s.content,
                    k.content
                ) AS content,

                COALESCE(
                    semantic_score,
                    0
                ) +

                COALESCE(
                    keyword_score,
                    0
                ) AS combined_score

            FROM semantic_search s

            FULL OUTER JOIN keyword_search k
            ON s.id = k.id

            ORDER BY combined_score DESC

            LIMIT :top_k
            """
        )

        with engine.begin() as conn:
            rows = conn.execute(
                sql,
                {
                    "embedding": query_embedding,
                    "query": query_text,
                    "top_k": top_k,
                },
            ).fetchall()

        return rows
```

# Production Reranker

## app/retrieval/reranker.py

```python id="m0gdn7"
from sentence_transformers import CrossEncoder


class Reranker:
    def __init__(self):
        self.model = CrossEncoder(
            "cross-encoder/ms-marco-MiniLM-L-6-v2"
        )

    def rerank(
        self,
        query,
        retrieved_docs,
        top_k=5,
    ):
        pairs = [
            [query, doc["content"]]
            for doc in retrieved_docs
        ]

        scores = self.model.predict(
            pairs
        )

        for doc, score in zip(
            retrieved_docs,
            scores,
        ):
            doc["rerank_score"] = float(score)

        reranked = sorted(
            retrieved_docs,
            key=lambda x: x["rerank_score"],
            reverse=True,
        )

        return reranked[:top_k]
```

# Interview-Level Concepts You Must Know

## Exact Search vs ANN

```text id="49qsmq"
Exact = accurate but slow
ANN = slightly approximate but very fast
```

---

# HNSW

Hierarchical graph-based ANN.

Very common in production.

---

# IVF

Clusters vectors first.

Speeds search further.

---

# Cosine Similarity

Measures vector angle similarity.

Central equation:

\text{cosine similarity}(A,B)=\frac{A\cdot B}{|A||B|}

Higher similarity:

```text id="zjxzbw"
more semantically related
```
